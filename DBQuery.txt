CREATE DATABASE Serotonia DEFAULT CHARACTER SET utf8;

USE Serotonia;

CREATE TABLE admin (
admin_id INTEGER NOT NULL AUTO_INCREMENT PRIMARY KEY,
email VARCHAR(128),
name VARCHAR(128),
password VARCHAR(128)
);

CREATE TABLE depressions (
depression_id INTEGER NOT NULL AUTO_INCREMENT PRIMARY KEY,
depression_level VARCHAR(50),
min INTEGER,
max INTEGER
);

CREATE TABLE persons_data (
person_id INTEGER NOT NULL AUTO_INCREMENT PRIMARY KEY,
age INTEGER,
gender VARCHAR(128),
depression_id INTEGER,
CONSTRAINT FOREIGN KEY (depression_id) REFERENCES depressions (depression_id)
ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE questions (
question_id INTEGER NOT NULL AUTO_INCREMENT PRIMARY KEY,
question TEXT
);

CREATE TABLE answers (
answer_id INTEGER NOT NULL AUTO_INCREMENT PRIMARY KEY,
answer TEXT,
rating INTEGER,
question_id INTEGER,
CONSTRAINT FOREIGN KEY (question_id) REFERENCES questions (question_id)
ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE persons_ans (
person_id INTEGER,
answer_id INTEGER,
PRIMARY KEY (person_id, answer_id),
CONSTRAINT FOREIGN KEY (person_id) REFERENCES persons_data (person_id)
ON DELETE CASCADE ON UPDATE CASCADE,
CONSTRAINT FOREIGN KEY (answer_id) REFERENCES answers (answer_id)
ON DELETE CASCADE ON UPDATE CASCADE
);
